import random

print ("Я задумал число от 1 до 20. Угадайте, какое это число, за минимум попыток!")
n=random.randint(1,20)

p=1
print("Попытка", p)
o=int(input("Введите число: "))
while o!=n:
    p+=1
    print ("Попытка", p)
    if n>o:
      print ("Моё число больше", o)
    else:
      print ("Моё число меньше", o)
    o=int(input("Попробуйте еще раз. Введите число: "))
print ("Поздравляю! Угадали! Это число",n)
print ("Количество попыток:", p)
